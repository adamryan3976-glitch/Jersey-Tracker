/** * Jersey Tracker Backend Script * * Paste this script directly inside Extensions -> Apps Script of your Google Sheet. * Ensure your Google Sheet contains three sheets (tabs): "Jerseys", "Out", and "In". * Once deployed as a Web App (Deploy -> New Deployment -> Web App), set the Access permission * to "Anyone" so the iPhone application can securely read and write data to your spreadsheets. */ function doGet(e) { var action = e.parameter.action; if (action === "getData") { return handleGetData(); } return ContentService.createTextOutput(JSON.stringify({ error: "Action not recognized" })) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var result = { success: false }; try { var requestData = JSON.parse(e.postData.contents); var action = requestData.action; var data = requestData.data; if (action === "assign") { result.success = handleAssign(data); } else if (action === "collect") { result.success = handleCollect(data); } else { result.error = "Unknown action requested"; } } catch (error) { result.error = error.toString(); } // Return standard success output with CORS allowance (GAS handles CORS automatically with TextOutput) return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function handleGetData() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var jerseysSheet = ss.getSheetByName("Jerseys"); var outSheet = ss.getSheetByName("Out"); var inSheet = ss.getSheetByName("In"); var payload = { jerseys: readSheetRows(jerseysSheet, ["sport", "number", "size"]), out: readSheetRows(outSheet, ["sport", "number", "size", "name", "date"]), in: readSheetRows(inSheet, ["sport", "number", "size", "name", "date"]) }; return ContentService.createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); } function handleAssign(data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var outSheet = ss.getSheetByName("Out"); if (!outSheet) return false; // Format dates consistently as ISO string timestamps var timestamp = new Date().toISOString(); outSheet.appendRow([ data.sport, data.number, data.size, data.name, timestamp ]); return true; } function handleCollect(dataList) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var inSheet = ss.getSheetByName("In"); if (!inSheet || !Array.isArray(dataList)) return false; var timestamp = new Date().toISOString(); dataList.forEach(function(item) { inSheet.appendRow([ item.sport, item.number, item.size, item.name, timestamp ]); }); return true; } /** * Generic helper to extract key-value row arrays from specified sheets. */ function readSheetRows(sheet, headers) { if (!sheet) return []; var lastRow = sheet.getLastRow(); if (lastRow < 2) return []; // Empty sheet or only contains headers var numCols = headers.length; var rangeValues = sheet.getRange(2, 1, lastRow - 1, numCols).getValues(); return rangeValues.map(function(row) { var obj = {}; headers.forEach(function(headerName, index) { obj[headerName] = row[index]; }); return obj; }); }
